# Repository
만든 프로젝트, 기타 공부 정보들을 모아둔 곳입니다.

# 자주 쓰는 Git 명령어들
```
$ git init // 로컬 저장소 생성
$ git remote add origin [원격저장소 주소] // 원격 저장소와 연결
$ git branch -m master main // 브랜치 이름 바꾸기
$ git branch [브랜치 이름] // 브랜치 만들기
$ git checkout 브랜치이름 // 브랜치로 이동하기
$ git add 파일명 // 파일 추가
$ git add . // 파일 전체 추가
$ git commit -m "설명" // 커밋할 때 설명 추가
$ git push origin main // 푸시할 떄
$ git pull origin main // 풀 할때
$ git clone [원격저장소 주소] // 원격저장소 로컬저장소에 복사
$ git config --global core.autocrlf true // git add 전 LF will be replaced by CRLF 에러 해결
```
